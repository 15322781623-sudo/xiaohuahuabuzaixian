# 批量推图功能代码实现文档

## 一、项目概述

本项目是一个基于 Vue 3 + Vite 构建的 XYZW 游戏管理系统（咸鱼之王游戏辅助工具），核心功能包括：
- **批量推图**（自动关卡推进）
- **十殿组队**（多人组队副本）
- **批量日常任务**（每日任务自动执行）
- **Token 管理**（多账号管理、WebSocket 通信）

## 四、战斗流程详解

### 4.1 核心战斗循环（`_bpPushOnce`）

这是推图的核心逻辑，每次执行一个完整的关卡战斗流程：

```
┌─────────────────────────────────────────────┐
│              _bpPushOnce(tid)               │
│                                             │
│  1. 获取角色信息 → 当前关卡ID               │
│     └─ role_getroleinfo                     │
│                                             │
│  2. 计算战斗时间                             │
│     └─ fight_calcleveltime → battleTime     │
│                                             │
│  3. 倒计时等待（战斗进行中）                  │
│     └─ 每秒更新countdown                     │
│     └─ 每25秒发送心跳                        │
│     └─ 每5秒刷新UI                          │
│                                             │
│  4. 获取战斗结果                             │
│     └─ fight_level → success/isWin          │
│                                             │
│  5. 判断胜负                                 │
│     ├─ 胜利: wins++, level=nextLevel         │
│     └─ 失败: losses++, retries++             │
│        ├─ retries >= 3 → 停止推图            │
│        └─ retries < 3 → 等待10秒重试         │
│                                             │
│  6. 刷新角色信息                             │
│     └─ role_getroleinfo                      │
└─────────────────────────────────────────────┘
```

**详细代码逻辑：**

```javascript
const _bpPushOnce = async (tid) => {
    var st = window._pt[tid];
    if (!st || st.stopFlag) return;

    // ===== 步骤1: 获取当前关卡 =====
    try {
        var ri = await r.sendMessageWithPromise(tid, "role_getroleinfo", {}, 10000);
        if (ri && ri.role) st.level = ri.role.levelId || 0;
    } catch (e) { /* 忽略，使用缓存的level */ }

    var _bn = _getBoss(st.level);
    _bpLog(tid, "当前关卡: " + st.level + (_bn ? " Boss: " + _bn : ""));
    _bpUpdCards();

    // ===== 步骤2: 计算战斗所需时间 =====
    var j = 300; // 默认300秒
    try {
        var cr = await r.sendMessageWithPromise(tid, "fight_calcleveltime", {}, 15000);
        if (cr && !cr.code) {
            var bt = cr.battleTime || (cr.body && cr.body.battleTime);
            if (bt != null) {
                j = Number(bt);
                if (j <= 0) j = 300;
            }
        }
        _bpLog(tid, "战斗需 " + j + " 秒", "success");
    } catch (e) {
        _bpLog(tid, "获取时间失败", "warning");
    }

    if (st.stopFlag) return;
    st.totalTime = j;
    st.countdown = j;
    _bpUpdCards();
    _bpLog(tid, "等待中...");

    // ===== 步骤3: 倒计时等待 =====
    var t0 = Date.now(), hb = 0;
    while (st.countdown > 0 && !st.stopFlag) {
        await _bpSleep(1000);
        st.countdown = Math.max(0, Math.ceil(j - (Date.now() - t0) / 1000));
        hb++;
        if (hb % 25 === 0) {
            try { r.sendMessage(tid, "heart_beat"); } catch (e) { }
        }
        if (hb % 5 === 0) _bpUpdCards();
    }

    if (st.stopFlag) return;

    // ===== 步骤4: 获取战斗结果 =====
    _bpLog(tid, "获取结果...");
    try {
        var fr = await r.sendMessageWithPromise(tid, "fight_level", {}, 15000);
        var bd = fr && fr.body || fr;
        var win = bd && (bd.success || bd.isWin) || false;
        var nl = bd && (bd.currLevel || bd.nextLevel) || st.level;

        // ===== 步骤5: 判断胜负 =====
        if (win) {
            st.wins++;
            st.retries = 0;
            st.level = nl;
            _bpLog(tid, "✅ 成功! 关卡 " + nl, "success");
        } else {
            st.losses++;
            st.retries = (st.retries || 0) + 1;
            _bpLog(tid, "❌ 失败 (Code:" + (bd && bd.code || "") + ")", "error");

            if (st.retries >= 3) {
                _bpLog(tid, "连续失败3次,停止", "error");
                st.stopFlag = true;
            } else {
                _bpLog(tid, "重试...", "warning");
                await _bpSleep(10000); // 等待10秒后重试
            }
        }

        // ===== 步骤6: 刷新角色信息 =====
        try {
            await r.sendMessageWithPromise(tid, "role_getroleinfo", {}, 8000);
        } catch (e) { }
        _bpUpdCards();

    } catch (e) {
        st.losses++;
        st.retries = (st.retries || 0) + 1;
        _bpLog(tid, "结果失败", "error");
        if (st.retries >= 3) st.stopFlag = true;
        else { await _bpSleep(10000); }
    }
    _bpUpdCards();
};
```

### 4.2 推图启动流程（`_bpStartOne`）

```javascript
const _bpStartOne = async (tid) => {
    // 1. 检查是否已在运行
    if (window._pt[tid] && window._pt[tid].running) return;

    // 2. 确保WebSocket已连接
    if (r.getWebSocketStatus(tid) !== "connected") {
        _bpLog(tid, "正在连接...", "info");
        try {
            var _tk = r.gameTokens.find(x => x.id === tid);
            if (_tk) r.selectToken(tid);
            await _bpSleep(3000);
        } catch (e) { }

        if (r.getWebSocketStatus(tid) !== "connected") {
            _bpLog(tid, "连接失败,跳过", "error");
            return;
        }
    }

    // 3. 初始化状态
    window._pt[tid] = {
        running: true,
        stopFlag: false,
        level: 0,
        wins: 0,
        losses: 0,
        retries: 0,
        countdown: 0,
        totalTime: 0
    };

    _bpLog(tid, "开始推图", "success");
    _bpUpdCards();

    // 4. 使用火把（如果启用）
    if (_bpTorch.value) {
        await _bpUseTorch(tid);
    }

    // 5. 主循环：不断执行战斗
    try {
        while (!window._pt[tid].stopFlag) {
            await _bpPushOnce(tid);          // 执行一次战斗
            if (window._pt[tid].stopFlag) break;

            // 火把续期检查
            if (_bpTorch.value && window._pt[tid].torchAt) {
                var _te = (Date.now() - window._pt[tid].torchAt) / 1000;
                if (_te >= window._pt[tid].torchDur) {
                    _bpLog(tid, "火把已过期,续用...", "warning");
                    await _bpUseTorch(tid);
                }
            }
            await _bpSleep(2000); // 战斗间隔2秒
        }
    } catch (e) {
        _bpLog(tid, "异常:" + e.message, "error");
    } finally {
        window._pt[tid].running = false;
        window._pt[tid].countdown = 0;
        _bpLog(tid, "已停止", "warning");
        _bpUpdCards();
    }
};
```

### 4.3 火把系统

```javascript
const _bpUseTorch = async (tid) => {
    var ti = _bpTorch.value;
    if (!ti) return;

    // 火把类型映射
    var nm = ti === 1008 ? "木材"
         : ti === 1009 ? "青铜"
         : "战神";

    _bpLog(tid, "使用" + nm + "火把 x10...", "info");

    var ok = 0;
    for (var i = 0; i < 10; i++) {
        try {
            await r.sendMessageWithPromise(tid, "item_consume", {
                itemId: ti,
                quantity: 1
            }, 5000);
            ok++;
            await _bpSleep(500);
        } catch (e) {
            _bpLog(tid, "火把第" + (i + 1) + "次失败: " + e.message, "error");
            break;
        }
    }

    // 记录火把激活时间和持续时长
    if (window._pt[tid]) {
        window._pt[tid].torchAt = Date.now();
        // 木材10min, 青铜20min, 战神30min (每个)
        window._pt[tid].torchDur = (ti === 1008 ? 600 : ti === 1009 ? 1200 : 1800) * ok;
    }

    var mins = Math.round((ti === 1008 ? 10 : ti === 1009 ? 20 : 30) * ok);
    _bpLog(tid, nm + "火把已激活 " + ok + "个(约" + mins + "分钟)", "success");
};
```

```javascript
// localTokenManager 中的连接创建
const createWebSocketConnection = async (roleId, token, wsUrl) => {
    // 1. 解析Token（支持Base64编码）
    let actualToken = token;
    try {
        const decoded = atob(token.replace(/^data:.*base64,/, "").trim());
        const parsed = JSON.parse(decoded);
        actualToken = parsed.token || parsed.gameToken || decoded;
    } catch (e) { /* 使用原始token */ }

    // 2. 创建WsAgent实例
    const agent = new WsAgent({
        heartbeatInterval: 2000,
        queueInterval: 50,
        channel: "x",
        autoReconnect: true,
        maxReconnectAttempts: 5
    });

    // 3. 连接成功回调
    agent.onOpen = () => {
        // 发送角色信息请求和数据版本请求
        agent.send(gameCommands.role_getroleinfo(0, 0, { roleId }));
        agent.send(gameCommands.system_getdatabundlever());

    // 4. 构建WebSocket URL
    const url = wsUrl || WsAgent.buildUrl("wss://xxz-xyzw.hortorgames.com/agent", {
        p: actualToken,
        e: "x",
        lang: "chinese"
    });

    // 5. 建立连接
    await agent.connect(url);
```

```javascript
// 竞技场战斗流程
async function arenaFight(tokenId, settings) {
    // 1. 时间检查（6:00-22:00）
    var hour = new Date().getHours();
    if (hour < 6 || hour > 22) return;

    // 2. 切换阵容
    await switchToFormationIfNeeded(tokenId, settings.arenaFormation, "竞技场阵容");

    // 3. 开始竞技场
    await executeGameCommand(tokenId, "arena_startarea", {});

    // 4. 执行3场战斗
    for (var l = 1; l <= 3; l++) {
        // 获取对手
        var target = await executeGameCommand(tokenId, "arena_getareatarget", {});
        var targetId = extractTargetId(target);

        // 开始战斗
        if (targetId) {
            await executeGameCommand(tokenId, "fight_startareaarena", {
                targetId: targetId
            }, 10000);
        }
        await sleep(1000);
    }
}
```

## 十二、技术要点总结

1. **BON 二进制协议**：所有游戏命令的 body 使用 BON 编码（自定义二进制格式），通过 `g_utils.bon.encode/decode` 处理
2. **WebSocket 消息队列**：命令先进入发送队列，由定时器（50ms间隔）批量处理发送
3. **Promise 响应匹配**：通过 `respKey` 将请求与响应匹配，支持超时机制（默认8秒）
4. **心跳保活**：每2秒发送心跳，维持连接
5. **自动重连**：断线后自动重连，最多5次，间隔3秒
6. **多账号并行**：支持同时选择多个账号进行批量推图，每个账号独立状态管理
7. **火把加速**：可选使用火把加速推图（木材/青铜/战神），自动续期
